#include <iostream>
#include <vector>
#include <string>

template<typename T>
class ExampleClass {
public:
    ExampleClass(const std::string& name) : name_(name) {}
    
    void processData(const std::vector<T>& data) {
        // 处理数据
        for (const auto& item : data) {
            std::cout << "Processing: " << item << std::endl;
        }
    }
    
private:
    std::string name_;
    int value_{42};
};

int main() {
    ExampleClass<int> example("test");
    std::vector<int> numbers = {1, 2, 3, 4, 5};
    example.processData(numbers);
    return 0;
}
